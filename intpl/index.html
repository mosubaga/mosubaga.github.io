Starter Script: Structured Output

This script formats a report from multiple arrays using references.

#!/usr/bin/perl
use strict;
use warnings;

my @languages = ("Perl", "Python", "Ruby");
my @levels    = ("Intermediate", "Intermediate", "Beginner");

my $rows = [\@languages, \@levels];

for my $i (0 .. $#{$rows->[0]}) {
    printf "%s - %s\n", $rows->[0][$i], $rows->[1][$i];
}
