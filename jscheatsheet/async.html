Asynchronous JavaScript

In the world of JavaScript, things don't always happen one after another. Imagine an ice cream shop where the server stands perfectly still while the freezer chills the gelato—that's synchronous code! Asynchronous JavaScript is what allows our "server" to take other orders or chat with customers while waiting for time-consuming tasks to finish. Whether you're fetching data from an API or waiting for a timer, mastering async patterns is essential for creating smooth, responsive web experiences that never keep your users waiting.

Handling tasks that take time (like fetching data or waiting for a timer) without freezing the browser.

1. Callback Functions

The "old school" way: passing a function as an argument to be executed later.


function orderIceCream(flavor, callback) {
  console.log(`Preparing ${flavor} ice cream...`);
  setTimeout(() => {
    callback(flavor);
  }, 2000);
}

orderIceCream('Chocolate', (item) => {
  console.log(`${item} is ready! 🍦`);
});

2. Promises

A Promise represents a value that might be available now, later, or never. It can be pending, fulfilled, or rejected.


const getScoops = new Promise((resolve, reject) => {
  const stockAvailable = true;
  if (stockAvailable) {
    resolve("Here are your scoops! 🍧");
  } else {
    reject("Sorry, out of stock! ❌");
  }
});

getScoops
  .then(message => console.log(message))
  .catch(error => console.error(error));

3. Async / Await

Syntactic sugar built on top of Promises, making asynchronous code look and behave more like synchronous code.


async function serveCustomer() {
  try {
    console.log("Taking order...");
    const result = await getScoops; // Wait for the promise to resolve
    console.log(result);
    return "Happy customer!";
  } catch (err) {
    console.log("Customer left unhappy: " + err);
  }
}

serveCustomer();

4. Fetch API

The Fetch API is the modern way to make HTTP requests in the browser. It returns a Promise that resolves to a Response object.

Basic GET request


// Fetch user data from an API
async function getUser(id) {
  const res = await fetch(`https://api.example.com/users/${id}`);

  // Always check if the request was successful
  if (!res.ok) {
    throw new Error(`HTTP error! Status: ${res.status}`);
  }

  const user = await res.json();
  console.log(user.name);
  return user;
}

getUser(1);

POST request with JSON body


// Create a new user
async function createUser(userData) {
  const res = await fetch('https://api.example.com/users', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': 'Bearer YOUR_TOKEN'
    },
    body: JSON.stringify(userData)
  });

  if (!res.ok) {
    const errorBody = await res.text();
    throw new Error(`Failed to create user: ${errorBody}`);
  }

  return await res.json();
}

createUser({ name: 'Alice', email: 'alice@example.com' });

Reusable fetch wrapper with error handling


// A handy wrapper you can reuse across your app
async function api(endpoint, options = {}) {
  const base = 'https://api.example.com';
  const res = await fetch(`${base}${endpoint}`, {
    headers: { 'Content-Type': 'application/json', ...options.headers },
    ...options
  });

  if (!res.ok) {
    throw new Error(`API ${res.status}: ${res.statusText}`);
  }

  return res.json();
}

// Usage
async function loadData() {
  const users = await api('/users');
  const newPost = await api('/posts', {
    method: 'POST',
    body: JSON.stringify({ title: 'Hello World' })
  });

  console.log(users, newPost);
}

5. Running Promises in Parallel

When you have multiple async operations that don't depend on each other, run them in parallel for better performance.

Promise.all — waits for all promises; fails immediately if any one rejects.


// Fetch user profile and their posts at the same time
async function loadDashboard(userId) {
  try {
    const [user, posts, notifications] = await Promise.all([
      fetch(`/api/users/${userId}`).then(r => r.json()),
      fetch(`/api/users/${userId}/posts`).then(r => r.json()),
      fetch(`/api/users/${userId}/notifications`).then(r => r.json())
    ]);

    console.log(`${user.name} has ${posts.length} posts`);
    console.log(`${notifications.length} unread notifications`);
  } catch (err) {
    console.error('One of the requests failed:', err);
  }
}

Promise.allSettled — waits for all promises and reports the status of each, never short-circuits.


// Try to send notifications to multiple services
async function notifyAll(data) {
  const results = await Promise.allSettled([
    fetch('/api/notify/email', { method: 'POST', body: data }),
    fetch('/api/notify/sms',   { method: 'POST', body: data }),
    fetch('/api/notify/slack', { method: 'POST', body: data })
  ]);

  results.forEach((result, i) => {
    if (result.status === 'fulfilled') {
      console.log(`Service ${i}: sent successfully`);
    } else {
      console.warn(`Service ${i}: failed -`, result.reason);
    }
  });
}

notifyAll(JSON.stringify({ message: 'Hello team' }));

Promise.race — resolves or rejects as soon as the first promise settles.


// Add a timeout to any fetch request
function fetchWithTimeout(url, ms = 5000) {
  const timeout = new Promise((_, reject) =>
    setTimeout(() => reject(new Error('Request timed out')), ms)
  );

  return Promise.race([fetch(url), timeout]);
}

async function loadWithTimeout() {
  try {
    const res = await fetchWithTimeout('/api/slow-endpoint', 3000);
    const data = await res.json();
    console.log(data);
  } catch (err) {
    console.error(err.message); // "Request timed out" if > 3s
  }
}

loadWithTimeout();

6. Common Async Patterns

Loading data on page load


// Load and display data when the page is ready
document.addEventListener('DOMContentLoaded', async () => {
  const container = document.getElementById('user-list');

  try {
    container.textContent = 'Loading...';
    const res = await fetch('/api/users');
    const users = await res.json();

    container.innerHTML = users
      .map(u => `
  • ${u.name} (${u.email})
    • `)
      .join('');
  } catch (err) {
    container.textContent = 'Failed to load users.';
    console.error(err);
  }
});

    Sequential vs Parallel fetching

    
// SLOW: Sequential - each request waits for the previous one
async function loadSequential() {
  const user  = await fetch('/api/user').then(r => r.json());   // ~300ms
  const posts = await fetch('/api/posts').then(r => r.json());  // ~300ms
  const stats = await fetch('/api/stats').then(r => r.json());  // ~300ms
  // Total: ~900ms
  return { user, posts, stats };
}

// FAST: Parallel - all requests start at the same time
async function loadParallel() {
  const [user, posts, stats] = await Promise.all([
    fetch('/api/user').then(r => r.json()),   // ~300ms
    fetch('/api/posts').then(r => r.json()),  // ~300ms (runs at same time)
    fetch('/api/stats').then(r => r.json())   // ~300ms (runs at same time)
  ]);
  // Total: ~300ms
  return { user, posts, stats };
}

    Simple retry pattern

    
// Retry a fetch request up to N times before giving up
async function fetchWithRetry(url, retries = 3, delay = 1000) {
  for (let attempt = 1; attempt <= retries; attempt++) {
    try {
      const res = await fetch(url);
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      return await res.json();
    } catch (err) {
      console.warn(`Attempt ${attempt} failed: ${err.message}`);
      if (attempt === retries) throw err;
      await new Promise(r => setTimeout(r, delay * attempt));
    }
  }
}

// Usage
async function loadFlakyData() {
  const data = await fetchWithRetry('/api/flaky-endpoint');
  console.log(data);
}
